#   posix, random, readline, Reflection, session, SimpleXML, sockets,
#   sodium, SPL, standard, sysvmsg, sysvsem, sysvshm, tokenizer, uri,
#   xml, xmlreader, xmlwriter, Zend OPcache, zlib.
#
# Lines starting with # are comments. Blank lines are ignored.
